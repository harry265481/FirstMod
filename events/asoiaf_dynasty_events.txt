			house = {
				scope:recipient = { set_house = house:house_Bar_Emmon }

						any_ancestor = {
							this = scope:recipient
						}
					}
					set_house = house:house_Bar_Emmon

			asoiaf_fertility_health_modifier_effect = yes
			
			#refresh CoAs on title shields
			dynamic_coa_effect = yes
		}
		#Can only happen once
