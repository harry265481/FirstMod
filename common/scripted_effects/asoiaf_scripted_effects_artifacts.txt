create_artifact_robert_warhammer_effect = {
	# Get the character the artifact is being made for.
	$OWNER$ = { save_scope_as = owner }

	set_artifact_rarity_famed = yes

	# Create the artifact
	create_artifact = {	
		name = robert_warhammer_name
		description = robert_warhammer_description
		type = hammer
		visuals = robert_warhammer_visuals
		wealth = scope:wealth
		quality = scope:quality
		template = robert_warhammer_template
		history = {
			type = created_before_history
		}
		modifier = robert_warhammer_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}

	scope:newly_created_artifact = {
		set_variable = { name = historical_unique_artifact value = yes }
	}
	
	scope:owner.house = {
		add_house_artifact_claim = scope:newly_created_artifact
	}
}

	$OWNER$ = { save_scope_as = owner }

	set_artifact_rarity_famed = yes

	# Create the artifact
	create_artifact = {	
		name = maekar_I_crown_name
		description = maekar_I_crown_description
		type = helmet
		visuals = maekar_I_crown_visuals
		wealth = scope:wealth
		quality = scope:quality
		template = maekar_I_crown_template
		history = {
			type = created_before_history
		}
		modifier = maekar_I_crown_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}

	scope:newly_created_artifact = {
		set_variable = { name = historical_unique_artifact value = yes }
	}
	
	scope:owner.house = {
		add_house_artifact_claim = scope:newly_created_artifact
	}
}
